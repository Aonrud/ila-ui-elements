A scroller applying some configuration settings and changing the number of items shown at each breakpoint.

new ila.Scroller(
	document.querySelectorAll(".scroller")[1],
	{
		texts: {
			left: '',
			right:'',
		},
		icons: {
			left: "iconoir-nav-arrow-left",
			right: "iconoir-nav-arrow-right",
		},
		titles: {
			left: "",
			right: ""
		},
		breakpoints: [ [0, 2], [768, 4], [992, 6], [1200, 6] ]
	}
);

Image Viewer

The image viewer on this page is instantiated with the button text symbols replaced with icons, and the close button title altered.

const iv = new ila.ImageViewer({
	showDownload: true,
	panzoom: true,
	texts: {
		cue: "",
		hide: "",
		download: "",
		prev: "",
		next: "",
		link: "",
		zoom: "",
		zoomActive: "",
	},
	icons: {
		cue: "iconoir-add-circled-outline",
		hide: "iconoir-delete-circled-outline",
		download: "iconoir-download-circled-outline",
		prev: "iconoir-arrow-left-circled",
		next: "iconoir-arrow-right-circled",
		link: "iconoir-link",
		zoom: "iconoir-expand",
		zoomActive: "iconoir-collapse",
	},
	titles: {
		hide: "Close this image"
	}
});
iv.create();
